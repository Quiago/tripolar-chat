```python
# model_manager/manager.py - Versión Enterprise
import os
import subprocess
import time
import requests
from typing import Optional, Dict, List
from pathlib import Path
import json

class VLLMModelManager:
    """
    Gestor profesional de modelos vLLM con:
    - Caché persistente de modelos HF
    - Hot-swap entre modelos
    - Métricas de uso
    """
    
    def __init__(self):
        self.current_model: Optional[str] = None
        self.process: Optional[subprocess.Popen] = None
        self.port = int(os.getenv("VLLM_PORT", "8000"))
        self.cache_dir = Path(os.getenv("HF_HOME", "/cache/huggingface"))
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        
        # Catálogo de modelos optimizados
        self.catalog = {
            "llama-3.1-8b": {
                "repo": "meta-llama/Llama-3.1-8B-Instruct",
                "quantization": None,
                "tensor_parallel": 1,
                "gpu_memory_util": 0.85,
                "max_model_len": 8192,
                "description": "Meta Llama 3.1 8B - Balance general"
            },
            "Llama-2-7b-chat-hf": {
                "repo": "casperhansen/llama-3.1-8b-instruct-awq",
                "quantization": "awq",
                "tensor_parallel": 1,
                "gpu_memory_util": 0.90,
                "max_model_len": 8192,
                "description": "Llama 8B 4-bit - Más eficiente"
            },
            "mistral-7b-awq": {
                "repo": "TheBloke/Mistral-7B-Instruct-v0.2-AWQ",
                "quantization": "awq",
                "tensor_parallel": 1,
                "gpu_memory_util": 0.90,
                "max_model_len": 8192,
                "description": "Mistral 7B - Bueno para código"
            },
            "qwen-2.5-7b": {
                "repo": "Qwen/Qwen2.5-7B-Instruct",
                "quantization": None,
                "tensor_parallel": 1,
                "gpu_memory_util": 0.85,
                "max_model_len": 8192,
                "description": "Qwen 2.5 - Multilingüe"
            }
        }
    
    def is_cached(self, model_name: str) -> bool:
        """Verifica si el modelo ya está en caché local"""
        config = self.catalog.get(model_name)
        if not config:
            return False
        
        # Check si los archivos del modelo existen en caché
        model_cache = self.cache_dir / "hub" / config["repo"].replace("/", "--")
        return model_cache.exists() and any(model_cache.iterdir())
    
    def get_model_info(self) -> Dict:
        """Info del modelo actual para métricas"""
        return {
            "current_model": self.current_model,
            "is_ready": self.is_ready(),
            "cache_dir": str(self.cache_dir),
            "catalog": list(self.catalog.keys())
        }
    
    def is_ready(self) -> bool:
        try:
            r = requests.get(f"http://localhost:{self.port}/health", timeout=3)
            return r.status_code == 200
        except:
            return False
    
    def stop(self):
        """Detiene vLLM gracefulmente"""
        if self.process and self.process.poll() is None:
            print(f"🛑 Deteniendo {self.current_model}...")
            self.process.terminate()
            try:
                self.process.wait(timeout=20)
            except subprocess.TimeoutExpired:
                self.process.kill()
                self.process.wait()
            
            # Esperar liberación de VRAM
            time.sleep(5)
            
            # Verificar con nvidia-smi que GPU está libre
            try:
                import subprocess
                result = subprocess.run(["nvidia-smi"], capture_output=True, text=True)
                print("📊 GPU status después de stop:")
                print(result.stdout[-500:])  # Últimas líneas
            except:
                pass
                
        self.current_model = None
        self.process = None
        print("✅ GPU liberada")
    
    def start(self, model_name: str):
        """Inicia vLLM con el modelo especificado"""
        if model_name not in self.catalog:
            raise ValueError(f"Modelo '{model_name}' no disponible. Catálogo: {list(self.catalog.keys())}")
        
        config = self.catalog[model_name]
        
        print(f"\n🚀 {'Cargando' if not self.is_cached(model_name) else 'Recuperando de caché'}: {model_name}")
        print(f"   Repositorio: {config['repo']}")
        print(f"   Caché: {self.cache_dir}")
        
        # Construir comando vLLM optimizado
        cmd = [
            "python", "-m", "vllm.entrypoints.openai.api_server",
            "--model", config["repo"],
            "--port", str(self.port),
            "--host", "0.0.0.0",
            "--gpu-memory-utilization", str(config["gpu_memory_util"]),
            "--max-model-len", str(config["max_model_len"]),
            "--dtype", "auto",
            "--enable-prefix-caching",  # Optimización: caché de prefijos
            "--enable-chunked-prefill",  # Optimización: chunked prefill
        ]
        
        if config.get("quantization"):
            cmd.extend(["--quantization", config["quantization"]])
        
        # Variables de entorno para caché
        env = os.environ.copy()
        env["HF_HOME"] = str(self.cache_dir)
        env["TRANSFORMERS_CACHE"] = str(self.cache_dir)
        env["HF_HUB_ENABLE_HF_TRANSFER"] = "1"  # Transferencia rápida
        
        print(f"   Comando: {' '.join(cmd[:8])}...")
        
        self.process = subprocess.Popen(
            cmd,
            env=env,
            preexec_fn=os.setsid,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE
        )
        
        # Esperar con polling progresivo
        print("⏳ Esperando que vLLM esté listo...")
        start_time = time.time()
        while time.time() - start_time < 180:  # 3 minutos timeout
            if self.is_ready():
                self.current_model = model_name
                elapsed = time.time() - start_time
                print(f"✅ Listo en {elapsed:.1f}s")
                return True
            time.sleep(1)
            
            # Verificar si proceso falló
            if self.process.poll() is not None:
                stdout, stderr = self.process.communicate()
                print(f"❌ Error:\n{stderr.decode()[-1000:]}")
                raise RuntimeError("vLLM falló al iniciar")
        
        raise TimeoutError("Timeout esperando vLLM")
    
    def ensure_loaded(self, model_name: str):
        """Asegura que el modelo esté cargado, haciendo swap si es necesario"""
        if self.current_model == model_name and self.is_ready():
            return
        if self.current_model:
            self.stop()
        self.start(model_name)
    
    def forward_chat(self, messages: List[Dict], **kwargs) -> Dict:
        """Envía chat a vLLM y retorna respuesta"""
        if not self.is_ready():
            raise RuntimeError("vLLM no está listo")
        
        payload = {
            "model": self.catalog[self.current_model]["repo"],
            "messages": messages,
            "max_tokens": kwargs.get("max_tokens", 512),
            "temperature": kwargs.get("temperature", 0.7),
            "top_p": kwargs.get("top_p", 0.9),
            "stream": False
        }
        
        r = requests.post(
            f"http://localhost:{self.port}/v1/chat/completions",
            json=payload,
            timeout=300
        )
        r.raise_for_status()
        return r.json()
```

# Tablas de base de datos
class User(SQLModel, table=True):
    id: Optional[str] = Field(default_factory=lambda: str(uuid.uuid4()), primary_key=True)
    username: str = Field(index=True, unique=True)
    email: str = Field(unique=True)
    hashed_password: str
    api_key: str = Field(default_factory=lambda: str(uuid.uuid4()), unique=True)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    is_active: bool = True

class Chat(SQLModel, table=True):
    id: Optional[str] = Field(default_factory=lambda: str(uuid.uuid4()), primary_key=True)
    user_id: str = Field(foreign_key="user.id", index=True)
    title: str = Field(default="Nueva conversación")
    model_used: str
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: datetime = Field(default_factory=datetime.utcnow)

class Message(SQLModel, table=True):
    id: Optional[str] = Field(default_factory=lambda: str(uuid.uuid4()), primary_key=True)
    chat_id: str = Field(foreign_key="chat.id", index=True)
    role: str  # "user" o "assistant"
    content: str
    created_at: datetime = Field(default_factory=datetime.utcnow)

# Crear tablas
SQLModel.metadata.create_all(engine)

# Auth
security = HTTPBearer()
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def verify_api_key(credentials: HTTPAuthorizationCredentials = Security(security), db: Session = Depends(get_db)):
    """Verifica API key y retorna usuario"""
    api_key = credentials.credentials
    user = db.exec(select(User).where(User.api_key == api_key)).first()
    if not user:
        raise HTTPException(status_code=403, detail="API Key inválida")
    if not user.is_active:
        raise HTTPException(status_code=403, detail="Usuario desactivado")
    return user

# Lifespan events

@app.post("/v1/chat/completions")
async def chat_completion(
    request: dict,
    user: User = Depends(verify_api_key),
    db: Session = Depends(get_db)
):
    """
    Endpoint OpenAI-compatible para chat.
    - Carga el modelo si no está cargado
    - Guarda historial en BD
    - Rate limiting por usuario
    """
    model_name = request.get("model")
    messages = request.get("messages", [])
    
    # Validar modelo
    if model_name not in model_manager.catalog:
        raise HTTPException(status_code=400, detail=f"Modelo {model_name} no disponible")
    
    try:
        # Asegurar modelo cargado (swap si es necesario)
        model_manager.ensure_loaded(model_name)
        
        # Crear o recuperar chat
        chat_id = request.get("chat_id")
        if chat_id:
            chat = db.get(Chat, chat_id)
            if not chat or chat.user_id != user.id:
                chat = None
        else:
            # Crear nuevo chat
            chat = Chat(
                user_id=user.id,
                title=messages[0]["content"][:50] if messages else "Chat",
                model_used=model_name
            )
            db.add(chat)
            db.commit()
            db
